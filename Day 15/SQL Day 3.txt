-- Create the Students table
CREATE TABLE students (
    StudentID INT PRIMARY KEY,
    StudentName VARCHAR(50),
    Dept VARCHAR(10),
    City VARCHAR(50)
);

-- Create the Results table
CREATE TABLE results (
    ResultID INT PRIMARY KEY,
    Subject VARCHAR(50),
    Marks INT,
    StudentID INT
);

-- Inner Join
select s.StudentName, s.Dept, r.Subject, r.Marks
from Students s
inner join Results r 
on  s.StudentID = r.StudentID;

-- left join
select s.StudentName, s.Dept, r.Subject, r.Marks
from Students s
left join Results r 
on  s.StudentID = r.StudentID;

-- right join
select s.StudentName, s.Dept, r.Subject, r.Marks
from Students s
right join Results r 
on  s.StudentID = r.StudentID;

-- Full outer join
select s.StudentName, s.Dept, r.Subject, r.Marks
from Students s
left join Results r 
on  s.StudentID = r.StudentID
union
select s.StudentName, s.Dept, r.Subject, r.Marks
from Students s
right join Results r 
on  s.StudentID = r.StudentID;


-- SUBQUERIES
-- 1. Find the name of the student with StudentID 1 using a subquery.
SELECT StudentName 
FROM Students 
WHERE StudentID = (SELECT StudentID FROM Students WHERE StudentID = 1);

-- 2. List all results for the student named 'Arjun'.
SELECT * FROM Results 
WHERE StudentID = (SELECT StudentID FROM Students WHERE StudentName = 'Arjun');

-- 3. Find all students who are in the same city as 'Priya'.
SELECT StudentName 
FROM Students 
WHERE City = (SELECT City FROM Students WHERE StudentName = 'Priya');

-- 4. Find all students who scored more than the average marks of all students.
SELECT StudentName 
FROM Students 
WHERE StudentID IN (
    SELECT StudentID 
    FROM Results 
    WHERE Marks > (SELECT AVG(Marks) FROM Results)
);

-- 5. List the names of students who have actually appeared for at least one exam.
SELECT StudentName 
FROM Students 
WHERE StudentID IN (SELECT DISTINCT StudentID FROM Results);

-- 6 Find the students who got the maximum marks in any subject.
SELECT StudentName 
FROM Students 
WHERE StudentID = (SELECT StudentID FROM Results WHERE Marks = (SELECT MAX(Marks) FROM Results));

-- 7. List students who belong to the same department as the student who scored 92.
SELECT StudentName 
FROM Students 
WHERE Dept = (
    SELECT Dept 
    FROM Students 
    WHERE StudentID = (SELECT StudentID FROM Results WHERE Marks = 92)
);

-- 8. Find students who have NOT appeared for any exam (using NOT EXISTS).
SELECT StudentName 
FROM Students s
WHERE NOT EXISTS (
    SELECT 1 
    FROM Results r 
    WHERE r.StudentID = s.StudentID
);

-- 9. Find the subjects that were taken by at least one student from 'Bengaluru'.
SELECT DISTINCT Subject 
FROM Results 
WHERE StudentID IN (SELECT StudentID FROM Students WHERE City = 'Bengaluru');

-- 10. Display each student's name along with a column showing the highest marks they achieved (Scalar Subquery).
SELECT StudentName, 
    (SELECT MAX(Marks) FROM Results r WHERE r.StudentID = s.StudentID) AS HighestMark
FROM Students s;
